Distributions

genes = vector()
for (i in 1:ncol(x)){
  genes = c(genes, x[,i])
}
experimental.design = vector()
for (i in 1:ncol(x)){
  experimental.design = c(experimental.design, rep(colnames(x)[i], nrow(x)))
}
x.new = data.frame(cbind(genes, experimental.design))
x.new$genes <- as.numeric(x.new$genes)
x.new$genes <- log1p(x.new$genes)
ggplot(x.new, aes(x=experimental.design, y=genes)) + geom_violin() + theme(axis.text = element_text(size = 6)) 

ggplot(x.new, aes(x=experimental.design, y=genes)) + geom_boxplot() + theme(axis.text = element_text(size = 6)) 

x.new$exp <- NA
x.new[x.new$experimental.design %in% c("bulk.DMSO-Pre-96IFC", "bulk.DMSO-Pre-800IFC", "bulk.DMSO-Post-800IFC"),]$exp <- c("DMSO")
